def solution(record):
    dic={}
    for record:
        if order=='Enter' or 'Change':
            dic[아이디]=닉네임
    for record:
        if order=='Enter':
            dic[아이디]님이 들어왔습니다.
        elif order=='Leave':
            dic[아이디]님이 나갔습니다.
